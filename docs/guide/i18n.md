# 国际化 (i18n)

使用 `withConvert` 方法实现单位的本地化显示。

## 基础用法

```ts
import SmartUnit from 'smart-unit'

// 创建基础实例
const size = new SmartUnit(['B', 'KB', 'MB', 'GB'], {
  baseDigit: 1024,
})

// 添加中文转换
const zhSize = size.withConvert((unit) => {
  const map = {
    B: '字节',
    KB: 'KB',
    MB: 'MB',
    GB: 'GB',
  }
  return map[unit] || unit
})

console.log(zhSize.format(1536))  // => "1.5 千字节"
```

## 链式格式化国际化

```ts
const time = new SmartUnit(['ms', 1000, 's', 60, 'm', 60, 'h'])

// 中文版
const zhTime = time.withConvert((unit) => {
  const map = { ms: '毫秒', s: '秒', m: '分钟', h: '小时' }

console.log(zhTime.formatChain(3661000, ' '))  
// => "1 小时 1 分钟 1 秒"
```

## 关键点

::: tip withConvert 特性
`withConvert` 返回的新实例继承原实例的所有配置（如 `fractionDigits`、`useDecimal`），只是添加了单位转换函数。
:::

::: info 翻译文本
翻译文本可以来自任何地方：硬编码、i18n 库、JSON 文件等。smart-unit 不依赖特定的 i18n 方案。
:::

## 多语言支持

### 完整的 i18n 示例

```ts
class I18nFormatter {
  private baseSize: SmartUnit
  
  constructor() {
    this.baseSize = new SmartUnit(['B', 'KB', 'MB', 'GB', 'TB'], {
      baseDigit: 1024,
      fractionDigits: 2,
    })
  }

  getFormatter(locale: string) {
    const translations: Record<string, Record<string, string>> = {
      'en': {
        B: 'B',
        KB: 'KB',
        MB: 'MB',
        GB: 'GB',
        TB: 'TB',
      },
      'zh-CN': {
        B: '字节',
        KB: 'KB',
        MB: 'MB',
        GB: 'GB',
        TB: 'TB',
      },
      'zh-TW': {
        B: '位元組',
        KB: 'KB',
        MB: 'MB',
        GB: 'GB',
        TB: 'TB',
      },
      'ja': {
        B: 'B',
        KB: 'KB',
        MB: 'MB',
        GB: 'GB',
        TB: 'TB',
      },
      'ko': {
        B: 'B',
        KB: 'KB',
        MB: 'MB',
        GB: 'GB',
        TB: 'TB',
      },
    }

    const trans = translations[locale] || translations['en']
    
    return this.baseSize.withConvert((unit) => {
      return trans[unit] || unit
    })
  }
}

// 使用
const i18n = new I18nFormatter()

console.log(i18n.getFormatter('en').format(1536))     // => "1.50KB"
console.log(i18n.getFormatter('zh-CN').format(1536))  // => "1.50KB"
console.log(i18n.getFormatter('ja').format(1536))     // => "1.50KB"
```

## 实际应用场景

### React + i18next 集成

```tsx
import { useTranslation } from 'react-i18next'
import SmartUnit from 'smart-unit'

function FileSize({ bytes }: { bytes: number }) {
  const { t, i18n } = useTranslation()
  
  const size = useMemo(() => {
    const baseSize = new SmartUnit(['B', 'KB', 'MB', 'GB'], {
      baseDigit: 1024,
      fractionDigits: 2,
    })
    
    const translations = {
      en: { B: 'B', KB: 'KB', MB: 'MB', GB: 'GB' },
      zh: { B: '字节', KB: 'KB', MB: 'MB', GB: 'GB' },
    } as const
    
    const locale = i18n.language.startsWith('zh') ? 'zh' : 'en'
    const trans = translations[locale]
    
    return baseSize.withConvert((unit) => trans[unit] || unit)
  }, [i18n.language])
  
  return <span>{size.format(bytes)}</span>
}

// 使用
<FileSize bytes={1536} />  // 根据当前语言显示 "1.50KB" 或 "1.50KB"
```

### 时间单位的多语言

```ts
class TimeI18n {
  private baseTime = new SmartUnit(['ms', 1000, 's', 60, 'm', 60, 'h'])

  getFormatter(locale: string) {
    const translations: Record<string, Record<string, string>> = {
      en: { ms: 'ms', s: 's', m: 'min', h: 'h' },
      zh: { ms: '毫秒', s: '秒', m: '分钟', h: '小时' },
      ja: { ms: 'ミリ秒', s: '秒', m: '分', h: '時間' },
      ko: { ms: '밀리초', s: '초', m: '분', h: '시간' },
    }

    const trans = translations[locale] || translations.en
    
    return this.baseTime.withConvert((unit) => trans[unit] || unit)
  }

  formatRelative(timestamp: number, locale: string): string {
    const now = Date.now()
    const diff = now - timestamp
    const formatter = this.getFormatter(locale)
    
    if (diff < 60000) {
      return formatter.format(diff)
    } else if (diff < 3600000) {
      return formatter.format(diff)
    } else {
      return formatter.format(diff)
    }
  }
}

// 使用
const timeI18n = new TimeI18n()

console.log(timeI18n.getFormatter('en').format(90000))    // => "1.5min"
console.log(timeI18n.getFormatter('zh').format(90000))    // => "1.5 分钟"
console.log(timeI18n.getFormatter('ja').format(90000))    // => "1.5 分"
```

## 链式格式化的国际化

```ts
class LocalizedChainFormatter {
  private baseTime = new SmartUnit(['s', 60, 'm', 60, 'h', 24, 'd'])

  getChainFormatter(locale: string, separator: string = ' ') {
    const translations: Record<string, Record<string, string>> = {
      en: { s: 's', m: 'm', h: 'h', d: 'd' },
      zh: { s: '秒', m: '分', h: '时', d: '天' },
      'zh-full': { s: '秒', m: '分钟', h: '小时', d: '天' },
    }

    const trans = translations[locale] || translations.en
    const localizedTime = this.baseTime.withConvert((unit) => trans[unit] || unit)
    
    return {
      format: (seconds: number) => localizedTime.formatChain(seconds, separator),
      getChain: (seconds: number) => localizedTime.getChainUnit(seconds),
    }
  }
}

// 使用
const chainFormatter = new LocalizedChainFormatter()

console.log(chainFormatter.getChainFormatter('en').format(3665))
// => "1h 1m 5s"

console.log(chainFormatter.getChainFormatter('zh').format(3665))
// => "1 时 1 分 5 秒"

console.log(chainFormatter.getChainFormatter('zh-full').format(3665))
// => "1 小时 1 分钟 5 秒"
```

## 自定义格式化逻辑

```ts
class SmartI18nFormatter {
  private baseSize = new SmartUnit(['B', 'KB', 'MB', 'GB'], {
    baseDigit: 1024,
    fractionDigits: '0-2',
  })

  getFormatter(locale: string, options?: {
    showFullUnit?: boolean
    useAbbreviation?: boolean
  }) {
    const { showFullUnit = false, useAbbreviation = true } = options || {}
    
    const translations: Record<string, Record<string, string>> = {
      en: {
        B: 'B',
        KB: useAbbreviation ? 'KB' : 'Kilobyte',
        MB: useAbbreviation ? 'MB' : 'Megabyte',
        GB: useAbbreviation ? 'GB' : 'Gigabyte',
      },
      zh: {
        B: showFullUnit ? '字节' : 'B',
        KB: showFullUnit ? '千字节' : 'KB',
        MB: showFullUnit ? '兆字节' : 'MB',
        GB: showFullUnit ? '吉字节' : 'GB',
      },
    }

    const trans = translations[locale] || translations.en
    
    return this.baseSize.withConvert((unit) => trans[unit] || unit)
  }
}

// 使用
const smartFormatter = new SmartI18nFormatter()

console.log(smartFormatter.getFormatter('zh').format(1536))
// => "1.5KB"

console.log(smartFormatter.getFormatter('zh', { showFullUnit: true }).format(1536))
// => "1.5 千字节"

console.log(smartFormatter.getFormatter('en', { showFullUnit: true }).format(1536))
// => "1.5 Kilobyte"
```

## 注意事项

::: info withConvert 的特性
`withConvert` 返回的新实例会继承原实例的所有配置（如 `fractionDigits`、`useDecimal` 等），只是添加了单位转换函数。

```ts
const base = new SmartUnit(['B', 'KB', 'MB'], {
  baseDigit: 1024,
  fractionDigits: 2,
  useDecimal: true,
})

const localized = base.withConvert(unit => ` translated_${unit}`)
// localized 继承了所有配置
console.log(localized.fractionDigits)  // => 2
console.log(localized.useDecimal)      // => true
```
:::
